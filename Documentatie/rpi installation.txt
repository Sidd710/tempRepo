Network access share

apt-get install samba samba-common-bin

mkdir /home/pi/Atum
mkdir /home/pi/Atum/printqueue (zonder sudo)
mkdir /home/pi/Atum/Atum.Update
mkdir /home/pi/Atum/Logging
nano /etc/samba/smb.config
[printqueue$]
   comment= Atum print queue
   path=/home/pi/Atum/printqueue
   browseable=Yes
   writeable=Yes
   only guest=no
   create mask=0777
   directory mask=0777
   public=yes

[logging$]
   comment= Atum Logging
   path=/home/pi/Atum/Logging
   browseable=Yes
   writeable=no
   only guest=no
   create mask=0777
   directory mask=0777
   public=yes

[atum.update$]
   comment= Atum update
   path=/home/pi/Atum/Atum.Update
   browseable=Yes
   writeable=Yes
   only guest=no
   create mask=0777
   directory mask=0777
   public=yes

--dont forget to copy legacy xml 

smbpasswd -a pi (empty)
smbpasswd -an nobody (empty)
chmod -R a+rwx ./printqueue
chmod -R a+rwx ./Atum.Update
chmod -R a+rwx ./Logging
